Also handle the `nextTurn` event to update the tile instruction when a turn advances.
The server's `advanceTurn` emits `nextTurn` with `{ currentTurnIndex, currentPlayer, currentPlayerName, turnNumber, toPosition }`.
Add `toPosition` to the nextTurn emit in server.ts (via advanceTurn call — it already receives `toPosition` as `tileIndex` parameter).

Actually — simplest approach without modifying server.ts: store `boardTilesData` received on game start, then on `nextTurn` show the instruction for `toPosition`:

```typescript
let boardTilesData: Array<{type: string; name: string; description: string}> = [];
let myCurrentTileIndex = 0;

socket.on('gameStarted', (data: any) => {
  if (data.boardTiles) boardTilesData = data.boardTiles;
  // ...existing game started logic
});

socket.on('nextTurn', (data: any) => {
  // ...existing turn indicator update
  // Update tile instruction if this is MY turn resolving
  // We'll update the instruction when we receive the tile position from gameState
});

socket.on('gameState', (state: any) => {
  const me = state.players[mySocketId];
  if (!me) return;
  if (statMoneyEl) statMoneyEl.textContent = '$' + me.money.toLocaleString();
  if (statFameEl)  statFameEl.textContent  = String(me.fame);
  if (statHapEl)   statHapEl.textContent   = String(me.happiness);
  if (statHpEl)    statHpEl.textContent    = String(me.hp ?? 10);
  if (statDegreeEl) statDegreeEl.textContent = me.degree ?? 'None';
  if (statCareerEl) statCareerEl.textContent = me.career ?? 'None';
  // Update tile instruction from current position
  if (me.position !== undefined && boardTilesData.length > 0) {
    const tile = boardTilesData[me.position];
    if (tile && tileInstrEl && tileNameEl && tileTextEl) {
      tileInstrEl.style.display = 'block';
      tileNameEl.textContent = 'Currently on: ' + tile.name;
      tileTextEl.textContent = tile.description;
    }
  }
});
```
