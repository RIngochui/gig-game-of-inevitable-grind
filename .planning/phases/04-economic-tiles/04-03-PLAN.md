```typescript
  socket.on('nepotism-select', ({ chosenPlayerId }: { chosenPlayerId: string }) => {
    const roomCode = findRoomCodeBySocketId(socket.id);
    if (!roomCode) { socket.emit('error', { message: 'You are not in a room' }); return; }

    const room = getRoom(roomCode);
    if (!room) { socket.emit('error', { message: 'Room was deleted' }); return; }

    if (room.gamePhase !== GAME_PHASES.PLAYING) {
      socket.emit('error', { message: 'Game is not in progress' }); return;
    }

    if (room.turnPhase !== TURN_PHASES.TILE_RESOLVING) {
      socket.emit('error', { message: 'No nepotism pending' }); return;
    }

    const currentPlayerId = room.turnOrder[room.currentTurnIndex];
    if (socket.id !== currentPlayerId) {
      socket.emit('error', { message: 'Not your turn' }); return;
    }

    if (chosenPlayerId === socket.id) {
      socket.emit('error', { message: 'Cannot choose yourself as beneficiary' }); return;
    }

    const beneficiary = room.players.get(chosenPlayerId);
    if (!beneficiary) {
      socket.emit('error', { message: 'Chosen player not found in room' }); return;
    }

    const benefactor = room.players.get(socket.id)!;
    beneficiary.money += 500;

    io.to(roomCode).emit('tile-nepotism-completed', {
      benefactorName: benefactor.name,
      benefactorNewMoney: benefactor.money,
      beneficiaryName: beneficiary.name,
      beneficiaryNewMoney: beneficiary.money
    });

    advanceTurn(room, roomCode, socket.id, benefactor.name, 0, benefactor.position, benefactor.position, 'NEPOTISM');
    console.log(`[nepotism-select] ${benefactor.name} → ${beneficiary.name} (+$500) in ${roomCode}`);
  });
```

```typescript
    case 'UNION_STRIKE': {
      // ECON-08: average all players' money, redistribute equally to all
      // Atomic: collect all mutations first, then single broadcast
      const totalMoney = Array.from(room.players.values()).reduce((sum, p) => sum + p.money, 0);
      const playerCount = room.players.size;
      const equalShare = Math.floor(totalMoney / playerCount);

      // Apply equal share to all players (single pass, no intermediate broadcasts)
      const afterBalances: { name: string; newMoney: number }[] = [];
      for (const [, p] of room.players) {
        p.money = equalShare;
        afterBalances.push({ name: p.name, newMoney: p.money });
      }

      // Single broadcast AFTER all mutations complete
      io.to(roomCode).emit('tile-union-strike', {
        playerCount,
        totalMoney,
        equalShare,
        afterBalances
      });

      advanceTurn(room, roomCode, playerId, player.name, roll, fromPosition, tileIndex, 'UNION_STRIKE');
      break;
    }
```
